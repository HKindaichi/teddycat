<script>
  const canvas = document.getElementById("gameCanvas");
  const ctx = canvas.getContext("2d");
  const gameOverScreen = document.getElementById("game-over");
  const finalScoreElement = document.getElementById("final-score");
  const retryButton = document.getElementById("retry-button");
  const submitNameButton = document.getElementById("submit-name");
  const playerNameInput = document.getElementById("player-name");
  const leaderboardInput = document.getElementById("leaderboard-input");
  const leaderboardList = document.getElementById("leaderboard-list");
  const scoreElement = document.getElementById("score");
  const highScoreElement = document.getElementById("high-score");
  const chancesElement = document.getElementById("chances");

  let gameRunning = false;
  let intervalId = null;
  let gameStarted = false;
  let playerClicked = false;

  const clickableImage = new Image();
  clickableImage.src = "https://i.postimg.cc/vmcZqSq3/forgame.png";

  let circle = { x: 150, y: 250, radius: 50, speed: 1000 };
  let score = 0;
  let chances = 10;
  let highScore = 0;

  let leaderboard = []; // Leaderboard to store top 5 players

  function showStartMessage(isRetry = false) {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    ctx.font = "16px 'Press Start 2P', cursive";
    ctx.fillStyle = "black";
    ctx.textAlign = "center";

    const message = isRetry ? "Start the game" : "Click to start the game";
    ctx.fillText(message, canvas.width / 2, canvas.height / 2 - 10);
    gameStarted = false;
  }

  function randomizeCircle() {
    circle.x = Math.random() * (canvas.width - 2 * circle.radius) + circle.radius;
    circle.y = Math.random() * (canvas.height - 2 * circle.radius) + circle.radius;
  }

  function updateUI() {
    scoreElement.textContent = score;
    highScoreElement.textContent = highScore;
    chancesElement.textContent = chances;
  }

  canvas.addEventListener("click", (e) => {
    if (!gameStarted) {
      startGame();
      return;
    }

    if (!gameRunning) return;

    playerClicked = true;
    const rect = canvas.getBoundingClientRect();
    const clickX = e.clientX - rect.left;
    const clickY = e.clientY - rect.top;

    const dist = Math.sqrt((clickX - circle.x) ** 2 + (clickY - circle.y) ** 2);
    if (dist <= circle.radius) {
      score++;
      if (score > highScore) highScore = score;

      if (score % 3 === 0 && circle.speed > 200) {
        circle.speed -= 50;
      }

      clearInterval(intervalId);
      randomizeCircle();
      intervalId = setInterval(moveCircle, circle.speed);
    } else {
      chances--;
      if (chances <= 0) endGame();
    }
    updateUI();
  });

  function moveCircle() {
    if (!playerClicked) {
      chances--;
      if (chances <= 0) {
        endGame();
        return;
      }
    }
    playerClicked = false;
    randomizeCircle();
    clearInterval(intervalId);
    intervalId = setInterval(moveCircle, circle.speed);
    draw();
    updateUI();
  }

  function startGame() {
    score = 0;
    chances = 10;
    circle.radius = 50;
    circle.speed = 1000;
    gameRunning = true;
    gameStarted = true;
    randomizeCircle();
    intervalId = setInterval(moveCircle, circle.speed);
    draw();
    updateUI();
  }

  function endGame() {
    gameRunning = false;
    clearInterval(intervalId);
    finalScoreElement.textContent = score;
    gameOverScreen.style.display = "block";

    const qualifies = leaderboard.length < 5 || score > leaderboard[leaderboard.length - 1].score;
    if (qualifies) leaderboardInput.style.display = "block";
  }

  submitNameButton.addEventListener("click", () => {
    const playerName = playerNameInput.value.trim();
    if (playerName === "") return;

    leaderboard.push({ name: playerName, score });
    leaderboard.sort((a, b) => b.score - a.score);
    if (leaderboard.length > 5) leaderboard.pop();

    updateLeaderboard();
    leaderboardInput.style.display = "none";
    playerNameInput.value = "";
  });

  retryButton.addEventListener("click", () => {
    gameOverScreen.style.display = "none";
    showStartMessage(true);
    gameRunning = false;
  });

  function updateLeaderboard() {
    leaderboardList.innerHTML = leaderboard
      .map((entry) => `<li>${entry.name} - ${entry.score}</li>`)
      .join("");
  }

  function draw() {
    if (!gameRunning) return;

    ctx.clearRect(0, 0, canvas.width, canvas.height);
    ctx.drawImage(
      clickableImage,
      circle.x - circle.radius,
      circle.y - circle.radius,
      circle.radius * 2,
      circle.radius * 2
    );
    requestAnimationFrame(draw);
  }

  showStartMessage();
